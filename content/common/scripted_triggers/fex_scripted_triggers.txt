fex_has_hyper_relay = {
    OR = {
        has_megastructure = hyper_relay_ruined
        has_megastructure = hyper_relay
        has_megastructure = hyper_relay_restored
    }
}

fex_is_bespoke_fe_world = {
    OR = {
        solar_system = {
            is_capital_system = yes
        }
        has_planet_flag = fe_the_preserve
        has_planet_flag = fex_the_archives
        # The Archives dont have a flag in vanilla
        AND = {
            owner = {
                is_fallen_empire_materialist = yes
            }
            is_planet_class = pc_gaia
            NOT = {
                has_planet_flag = fex_experiment_world
            }
        }
    }
}

fex_is_exodite_country = {
    capital_scope = {
        has_planet_flag = fex_exodite_planet
    }
}

fex_can_raid_tombs = {
    any_deposit = {
        OR = {
            is_deposit_type = d_fex_crypt_monument
            is_deposit_type = d_fex_ancient_barrows
            is_deposit_type = d_fex_imperial_tombs
        }
        NOT = { has_deposit_flag = fex_plundered}
    }
}

fex_is_historical_border_system = {
    or = {
        fex_is_historical_border_system_type = { FE_TYPE = materialist}
        fex_is_historical_border_system_type = { FE_TYPE = spiritualist}
        fex_is_historical_border_system_type = { FE_TYPE = xenophile}
        fex_is_historical_border_system_type = { FE_TYPE = xenophobe}
    }
}

fex_is_historical_border_system_type = {
    AND = {
        has_star_flag = fex_$FE_TYPE$_history
        any_neighbor_system = {
            AND = {
                OR = {
                    has_star_flag = fex_materialist_history
                    has_star_flag = fex_spiritualist_history
                    has_star_flag = fex_xenophile_history
                    has_star_flag = fex_xenophobe_history
                }
                # This is to ensure that the neighbours are a different type
                NOT = {
                    has_star_flag = fex_$FE_TYPE$_history
                }
            }
        }
    }
}
# System scope
fex_is_suitable_system_for_special_discoverables = {
    NOR = {
        exists = owner
        any_system_planet = { is_colony = yes }
        has_any_megastructure = yes
        has_star_flag = fex_discoverable_system # No doubling up
    }
    system_star = {
        nor = {
            is_planet_class = pc_neutron_star
            is_planet_class = pc_rift_star
            is_planet_class = pc_black_hole
            is_planet_class = pc_pulsar
        }
    }
    any_system_planet = {
        fex_is_suitable_planet_for_special_discoverables = yes
    }
}

fex_is_suitable_planet_for_special_discoverables = {
    OR = {
        is_planet_class = pc_toxic
        is_planet_class = pc_barren_cold
        is_planet_class = pc_barren
        is_planet_class = pc_molten
    }
    NOR = {
        is_moon = yes # For some reason this breaks the custom image stuff.
        has_planet_flag = machine_lair
        has_planet_flag = machine_world_1
        has_planet_flag = machine_world_2
        has_planet_flag = machine_world_3
        has_planet_flag = machine_world_4
    }
}
