fex_xeno_game = {
	category = xeno_ward
	is_capped_by_modifier = no

	swappable_data = {
		default = {
			building_icon = building_crude_huts
    		icon = fe_xeno_ward
		}
	}

	possible_pre_triggers = {
		has_owner = yes
		is_sapient = yes
	}

	possible_precalc = can_fill_precursor_job

	possible = {
		owner = {
			is_fallen_empire_xenophobe = yes
		}
		planet = {
			has_planet_flag = fex_hunting_world
		}
		NOT = { is_same_species = owner }
		NOT = { has_trait = trait_nerve_stapled} # To avoid the slave species being put here.
		is_robot_pop = no
	}

	resources = {
		category = planet_jobs
		produces = {
			unity = 2
		}
	}

    triggered_planet_modifier = {
		potential = {
			always = yes
		}
		planet_amenities_add = 3
		mult = value:scripted_modifier_mult|MODIFIER|pop_job_amenities_mult|
	}

	weight = {
		weight = @worker_job_weight

		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
		}
	}
}
