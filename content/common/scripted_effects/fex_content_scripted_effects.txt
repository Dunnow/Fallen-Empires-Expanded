					set_location = {

					set_location = {
						target = prev

				}

		}
	}
	last_created_fleet = {

fex_create_shroud_incursion_fleet = {
	# Create Shroud Country
	if = {
		limit = { NOT = { exists = event_target:shroud_country } }
		create_country = {
			name = "NAME_Creatures_of_the_Shroud"
			type = shroud_spirits
			flag = {
				icon = {
					category = "special"
					file = "the_shroud.dds"
				}
				background= {
					category = "backgrounds"
					file = "00_solid.dds"
				}
				colors = {
					"dark_purple"
					"black"
					"null"
					"null"
				}
			}
		}
		last_created_country = {
			save_global_event_target_as = shroud_country
		}
	}
	create_fleet = {
		name = "NAME_Psionic_Entity"
		settings = {
			spawn_debris = no
			is_boss = yes
		}
		effect = {
			set_owner = event_target:shroud_country
			create_ship = {
				name = "NAME_Psionic_Entity"
				design = "NAME_Corrupted_Avatar"
			}
			set_location = {
				target = prev.star
				distance = 100
				angle = random
			}
			set_fleet_stance = aggressive
			set_aggro_range_measure_from = return_point
			set_aggro_range = 2000
		}
	}
}

# Leader scope - send a leader into exile for a pilgrimage.
fex_send_leader_on_pilgrimage = {
	remove_leader_flag = fex_pilgrim_candidate
	set_leader_flag = fex_on_pilgrimage
	unassign_leader = this
	exile_leader_as = fex_pilgrim_exile_@root
	save_global_event_target_as = fex_current_pilgrim_@root
}
