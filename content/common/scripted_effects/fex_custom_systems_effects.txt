fex_generate_exodites_on_planet = {
	optimize_memory
	set_planet_flag = pre_ftl_default
    set_planet_flag = fex_exodite_planet
	change_pc = pc_gaia
	reroll_planet = yes
	fex_create_exodite_country = yes
	fex_setup_exodite_planet = yes

	if = {
		limit = { is_moon = yes }
		orbit = {
			random_list = {
				1 = { set_name = fex_exodite_moon_1 }
				1 = { set_name = fex_exodite_moon_2 }
				1 = { set_name = fex_exodite_moon_3 }
				1 = { set_name = fex_exodite_moon_4 }
				1 = { set_name = fex_exodite_moon_5 }
				1 = { set_name = fex_exodite_moon_6 }
			}
		}
	}
}

fex_create_exodite_country = {
    create_country = {
		name = random
		authority = random
		civics = {
			civic = civic_landed_nobility
			civic = civic_flat_world_theory
		}
		species = event_target:fex_xenophile_fe.species
		ethos = {
            ethic = ethic_fanatic_xenophile
        }
		flag = {
			icon = {
				category = "pre_ftl"
				file = "preftl_stone_age.dds"
			}
			background= {
				category = "backgrounds"
				file = "new_dawn.dds"
			}
			colors={
				"turquoise"
				"green"
				"null"
				"null"
			}
		}
		origin = "origin_default"
		type = primitive
	}
    last_created_country = {
		set_graphical_culture = preindustrial_01
        set_pre_ftl_age = fex_exodite_age
        change_government = {
            authority = random
            civics = {
                civic = fex_civic_natural_harmony
                civic = fex_civic_archaeotech_legacy
            }
        }
		room_name_override = "fex_exodite_room"
    }
    set_pre_ftl_origin = yes
}

fex_setup_exodite_planet = {
	set_owner = last_created_country
	set_capital = yes

    remove_building = building_colony_shelter
    add_building = building_fex_exodite_settlements
	add_building = building_fex_runestones
    while = {
        count = 9
        create_pop = {
            species = event_target:fex_xenophile_fe.species
        }
    }
	set_name = random
}

fex_generate_xenophobe_exiles = {
	save_event_target_as = fex_xenophobe_exiles_gate_system
	set_star_flag = fex_xenophobe_exiles_gate
	spawn_system = {
		initializer = fex_black_hole_init
		hyperlane = no
	}
	last_created_system = {
		save_global_event_target_as = fex_xenophobe_exiles_system
		set_star_flag = fex_xenophobe_history
		set_star_flag = fex_historical_system
		set_star_flag = fex_xenophobe_exiles_system

		fex_generate_xenophobe_exile_habitat = yes
		fex_generate_xenophobe_exile_fleets = yes
	}

	event_target:fex_xenophobe_exiles_gate_system = {
		event_target:fex_xenophobe_exiles_system = {
			prev = {
				spawn_natural_wormhole = {
					bypass_type = wormhole
					random_pos = no
					orbit_angle = 360
				}
			}
			spawn_natural_wormhole = {
				bypass_type = wormhole
				random_pos = no
				orbit_angle = 270
			}
			link_wormholes = prev
		}
	}
}

# System scope
fex_generate_xenophobe_exile_habitat = {
	random_system_planet = {
		limit = {
			NOR = {
				is_star = yes
				is_moon = yes
			}
		}
		save_event_target_as = fex_exile_habitat_root
	}

	spawn_planet = {
		class = pc_habitat
		location = event_target:fex_exile_habitat_root
		init_effect = {
			set_planet_entity = {
				entity = fex_fallen_empire_habitat_entity
				picture = pc_fex_exile_hab
			}
			set_name = fex_exile_hab_name
			save_global_event_target_as = fex_exile_habitat
		}
	}
	event_target:fex_exile_habitat = {
		fex_create_exile_country = yes
		fex_setup_exile_planet = yes
	}

	create_starbase = {
		size = starbase_outpost
		owner = last_created_country
	}
}

fex_create_exile_country = {
	create_country = {
		type = fex_exiles_country
		species = event_target:fex_xenophobe_fe.species
		ethos = {
			ethic = ethic_fanatic_xenophobe
			ethic = ethic_militarist
		}
		civics = {
			civic = fex_civic_blood_exiles
		}
	}

	event_target:fex_xenophobe_fe = {
		set_faction_hostility = { set_neutral = yes }
	}

	last_created_country = {
		room_name_override = "personality_slaving_despots_room"
		set_graphical_culture = fallen_empire_04
		save_global_event_target_as = fex_xenophobe_exiles_country
		set_name = {
			key = fex_exile_country_name
			variable_string = "\\[This.owner_main_species.GetAdj]"
		}
	}
}

fex_setup_exile_planet = {
	set_owner = last_created_country
	set_capital = yes

	remove_building = building_colony_shelter
	add_building = building_fex_dark_palace
	add_building = building_fex_purity_temple
	add_building = building_fex_blood_arena
	add_building = building_fex_slave_pens

	add_deposit = d_arcane_replicator

    while = {
        count = 14
        create_pop = {
            species = event_target:fex_xenophobe_fe.species
        }
    }
	while = {
        count = 8
        random_country = {
            limit = {
                OR = {
                    is_country_type = default
                    is_primitive = yes
                }
                is_gestalt = no
            }
            species = {
                save_event_target_as = fex_slave_target
				set_citizenship_type = { country = event_target:fex_xenophobe_exiles_country type = citizenship_slavery cooldown = yes }
            }
        }
        create_pop = { species = event_target:fex_slave_target }
    }
}

fex_generate_xenophobe_exile_fleets = {
	# Generate home fleet
	fex_generate_exile_admiral = yes
	create_fleet = {
		name = "NAME_fex_exiles_home_fleet"
		effect = {
			set_owner = event_target:fex_xenophobe_exiles_country
			set_location = event_target:fex_xenophobe_exiles_country.capital_star
			set_fleet_flag = fex_exile_home_fleet

			while = {
				count = 4
				create_ship = {
					name = random
					design = NAME_Supremacy
				}
			}
			while = {
				count = 20
				create_ship = {
					name = random
					design = NAME_Glory
				}
			}
		}
		settings = {
			can_upgrade = no
			can_change_composition = no
			can_change_leader = no
			uses_naval_capacity = no
			spawn_debris = no
		}
	}
	last_created_fleet = { assign_leader = last_created_leader }
	# Generate Raiding parties
	fex_generate_exile_raiding_fleet = { NAME = NAME_fex_exiles_raiding_fleet_1 }
	fex_generate_exile_raiding_fleet = { NAME = NAME_fex_exiles_raiding_fleet_2 }
	fex_generate_exile_raiding_fleet = { NAME = NAME_fex_exiles_raiding_fleet_3 }
}
