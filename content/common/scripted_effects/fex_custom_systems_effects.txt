# Planet scope
fex_generate_exodites_on_planet = {
	optimize_memory
	set_planet_flag = pre_ftl_default
    set_planet_flag = fex_exodite_planet
	change_pc = pc_gaia
	reroll_planet = yes
	add_modifier = { modifier = fex_fe_exodite_terraforming }
	fex_create_exodite_country = yes
	fex_setup_exodite_planet = yes

	if = {
		limit = { is_moon = yes }
		orbit = {
			random_list = {
				1 = { set_name = fex_exodite_moon_1 }
				1 = { set_name = fex_exodite_moon_2 }
				1 = { set_name = fex_exodite_moon_3 }
				1 = { set_name = fex_exodite_moon_4 }
				1 = { set_name = fex_exodite_moon_5 }
				1 = { set_name = fex_exodite_moon_6 }
			}
		}
	}
}

fex_create_exodite_country = {
    create_country = {
		name = random
		authority = random
		civics = {
			civic = civic_landed_nobility
			civic = civic_flat_world_theory
		}
		species = event_target:fex_xenophile_fe.species
		ethos = {
            ethic = ethic_fanatic_xenophile
        }
		flag = {
			icon = {
				category = "pre_ftl"
				file = "preftl_stone_age.dds"
			}
			background= {
				category = "backgrounds"
				file = "new_dawn.dds"
			}
			colors={
				"turquoise"
				"green"
				"null"
				"null"
			}
		}
		origin = "origin_default"
		type = primitive
	}
    last_created_country = {
		set_graphical_culture = preindustrial_01
        set_pre_ftl_age = fex_exodite_age
        change_government = {
            authority = random
            civics = {
                civic = fex_civic_natural_harmony
                civic = fex_civic_archaeotech_legacy
            }
        }
		room_name_override = "fex_exodite_room"
    }
    set_pre_ftl_origin = yes
}

fex_setup_exodite_planet = {
	set_owner = last_created_country
	set_capital = yes

    remove_building = building_colony_shelter
    add_building = building_fex_exodite_settlements
	add_building = building_fex_runestones
    while = {
        count = 9
        create_pop = {
            species = event_target:fex_xenophile_fe.species
        }
    }
	set_name = random
}
