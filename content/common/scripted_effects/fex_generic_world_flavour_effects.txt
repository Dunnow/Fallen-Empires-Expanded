# All are planet scope

# Hunting world, Xenophobe
fex_set_hunting_world = {
    set_planet_flag = fex_hunting_world
    add_modifier = { modifier = fex_xenophobe_titanic_game }
    add_building = building_fex_hunting_lodges

    # Add primitive game
    while = {
        count = 5
        random_country = {
            limit = {
                OR = {
                    is_country_type = default
                    is_primitive = yes
                }
                is_gestalt = no
            }
            species = {
                save_event_target_as = fex_hunting_target
            }
        }
        create_pop = { species = event_target:fex_hunting_target }
    }
}
