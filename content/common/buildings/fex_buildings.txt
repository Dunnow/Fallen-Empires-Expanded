building_fex_hunting_lodges= {
	base_buildtime = @b1_time

	category = amenity

	base_cap_amount = 1

	building_sets = {
		fallen_empire
	}

	potential = {
		exists = owner
		owner = {
			is_fallen_empire_xenophobe = yes
		}
		has_planet_flag = fex_hunting_world
	}

	destroy_trigger = {
		exists = owner
		owner = {
			is_fallen_empire_xenophobe = no
		}
	}

	planet_modifier = {
		planet_housing_add = 300
		planet_amenities_add = 500
		job_fex_xeno_hunter_add = 300
		job_clerk_add = 300
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @b1_minerals
		}
		upkeep = {
			energy = 2
		}
	}
}

building_fex_exodite_settlements = {
	capital = yes
	can_build = no
	can_demolish = yes
	can_be_ruined = no
	position_priority = 0
	icon = holding_tree_of_life_sapling

	building_sets = {
		pre_ftl
	}

	category = government

	potential = {
		exists = owner
		owner = {
			is_primitive = yes
			is_hive_empire = no
		}
		has_planet_flag = fex_exodite_planet
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				is_primitive = no
				is_hive_empire = yes
			}
		}
	}

	convert_to = {
		building_machine_capital
		building_capital
		building_hive_capital
		building_hab_capital
		building_hab_fe_capital
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				is_primitive = yes
			}
			has_planet_flag = fex_exodite_planet
		}

		planet_housing_add = 1000
		planet_amenities_add = 2500
		planet_crime_add = -20

		job_fex_exodite_hunter_add = 300
		job_fex_exodite_smith_add = 300
	}
}
