		orbit_distance = 0

		has_ring = no
	}

		moon = {
			count = { min = 0 max = 1 }

		}

	init_effect = {

				type = "extradimensional_2"
				location = this
			}

				set_ambient_object_flag = extradimensional_system_effect_2
				set_location = {

					angle = random

		create_fe_citadel = yes
		starbase = {
			fex_create_shroud_tear_defense_fleet = yes
		}
		every_system_planet = {
			limit = {
				OR = {
					is_colonizable = yes
					is_planet_class = pc_barren
					is_planet_class = pc_barren_cold
					is_planet_class = pc_molten
					is_planet_class = pc_toxic
					is_planet_class = pc_frozen
				}
			}
			change_pc = pc_shrouded
			reroll_planet_modifiers = yes
			random_list = {
				1 = { add_deposit = d_zro_deposit_3 }
				1 = { add_deposit = d_zro_deposit_4}
				1 = { add_deposit = d_zro_deposit_5 }
			}
		}
		# Create stations
		every_system_planet = {
			create_research_station = { owner = starbase.owner }
			create_mining_station = { owner = starbase.owner }
		}
	}
