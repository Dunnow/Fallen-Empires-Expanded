fex_exodus_precursor = {
	rank = 1
	change_job_threshold = 2

    assign_to_pop = {
		exists = owner
		owner = {
			is_country_type = primitive
			has_pre_ftl_age = fex_exodite_age
		}
		owner = { is_hive_empire = no }
	}

	weight = {
		weight = 100

		# crisis purge
		modifier = {
			factor = 0.01
			exists = planet
			exists = planet.controller
			planet.controller = {
				OR = {
					is_country_type = swarm
					is_country_type = ai_empire
				}
			}
		}
		modifier = {
			factor = 0.01
			is_unemployed = yes
		}
	}

	pop_group_modifier = {
		pop_happiness = 0.25
		pop_political_power = 1
		pop_housing_usage_base = 1
		pop_amenities_usage_base = 1
	}

	# scope: pop to be resettled
	allow_resettlement = {
		always = no
	}

	resources = {
		category = pop_category_precursor

		# Consumer Goods Upkeep
		upkeep = {
			trigger = {
				exists = owner
				owner = {
					is_spiritualist = no
				}
			}
			consumer_goods = 2.0
		}

		upkeep = {
			trigger = {
				exists = owner
				owner = {
					is_spiritualist = yes
				}
			}
			consumer_goods = 1.0
		}
	}
}


fex_exile_precursor = {
    rank = 1
    clothes_texture_index = 2
    change_job_threshold = 2

    demotion_time = 0
    display_unemployment = no

    assign_to_pop = {
		exists = owner
		owner = {
			is_country_type = fex_exiles_country
		}
		owner = { is_hive_empire = no }
	}

    triggered_planet_modifier = {
        potential = {
            has_trait = trait_noxious
        }
        key = trait_noxious
        pop_non_noxious_happiness = @noxious_happiness_penalty
    }

    should_apply_unemployment_penalties = {
        NOR = {
            has_very_high_living_standard_type = yes
            has_high_living_standard_type = yes
        }
    }

    pop_group_modifier = {
        pop_happiness = 0.15
        pop_political_power = 1
        pop_housing_usage_base = 1
        pop_amenities_usage_base = 1
    }

    # scope: pop to be resettled
    allow_resettlement = {
        always = no
    }
}
