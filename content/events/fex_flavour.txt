}

	hide_window = yes
	is_triggered_only = yes

		exists = fromfrom

			is_primitive = yes
		}
		has_planet_flag = fex_exodite_planet

		event_target:fex_xenophile_fe = {
            establish_communications = from

		save_event_target_as = fex_invaded_exodite_planet
		from = {
            country_event = { id = fex_flavour.13 }
		}
	}
}

# Bounce from the planet
planet_event = {
    id = fex_flavour.14
	hide_window = yes
	is_triggered_only = yes

    immediate = {
		fex_create_exodite_country = yes
		set_owner = last_created_country
		set_capital = yes

		remove_modifier = planet_culture_shock
		remove_building = building_colony_shelter
		add_building = building_fex_exodite_settlements
		add_building = building_fex_runestones

		remove_all_districts = yes

		# Quietly recreate comms with countries so that this isn't lost
		every_country = {
			limit = {
				root = { is_surveyed  = { who = prev status = yes} }
			}
            establish_communications = last_created_country
		}

		last_created_country = {
            set_awareness = 100
		}
		planet_event = { id = fex_flavour.15 } # Regrow natives
		planet_event = { id = fex_flavour.16 } # Get rid of colonists
	}
}

planet_event = {
    id = fex_flavour.15
	hide_window = yes
	is_triggered_only = yes

    immediate = {

			limit = {
				exists = owner
				owner = {
					fex_is_exodite_country = yes
				}
				count_owned_pop_amount = {
					count < 900
					limit = {
						is_same_species = event_target:fex_xenophile_fe.species
					}
				}
			}
			create_pop = {
				species = event_target:fex_xenophile_fe.species
			}
			planet_event = { id = fex_flavour.15 days = 360 }
		}
	}
}

    id = fex_flavour.30

    desc = fex_flavour.30.desc
    picture = GFX_evt_excavation_team
	location = this
    is_triggered_only = yes

	immediate = {
		solar_system = {
			switch = {
				trigger = has_star_flag
				fex_xenophobe_history = { event_target:fex_xenophobe_fe = { save_event_target_as = fex_burial_origin_country} }
				fex_xenophile_history = { event_target:fex_xenophile_fe = { save_event_target_as = fex_burial_origin_country} }
				fex_materialist_history = { event_target:fex_materialist_fe = { save_event_target_as = fex_burial_origin_country} }
				fex_spiritualist_history = { event_target:fex_spiritualist_fe = { save_event_target_as = fex_burial_origin_country} }
				default = {
					random_country = {
						limit = {
							is_fallen_empire = yes
							is_fallen_empire_machine = no
						}
						save_event_target_as = fex_burial_origin_country
					}
				}
			}
		}
	}

	option = {
		name = fex_flavour.30.a
		owner = {
			add_resource = {
				minor_artifacts = 100
			}
			# 60 months of society
			add_monthly_resource_mult = {
				resource = society_research
				value = 60
				min = 200
				max = 9999
			}
			# 24 month of energy
			add_monthly_resource_mult = {
				resource = energy
				value = 24
				min = 200
				max = 9999
			}
			# 1 year of other research types
			add_monthly_resource_mult = {
				resource = physics_research
				value = 12
				min = 200
				max = 9999
			}
			add_monthly_resource_mult = {
				resource = engineering_research
				value = 12
				min = 200
				max = 9999
			}
		}
	}
}

system_event = {
	id = fex_flavour.45
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_system_ambient_object = {
			limit = {
				is_ambient_object_type = extradimensional_2
			}
			destroy_ambient_object = this
		}

		every_fleet_in_system = {
			limit = {
				owner = { is_country_type = shroud_spirits}
				any_owned_ship = {
					is_ship_size = corrupted_avatar
				}
			}
			delete_fleet = this
		}

		every_system_planet = {
			switch = {
				trigger = has_deposit
				d_zro_deposit_1 = {
					remove_deposit = d_zro_deposit_1
				}
				d_zro_deposit_2 = {
					remove_deposit = d_zro_deposit_2
				}
				d_zro_deposit_3 = {
					remove_deposit = d_zro_deposit_3
					random_list = {
						1 = {}
						1 = { add_deposit = d_zro_deposit_1}
					}
				}
				d_zro_deposit_4 = {
					remove_deposit = d_zro_deposit_4
					random_list = {
						1 = {}
						1 = { add_deposit = d_zro_deposit_1}
						1 = { add_deposit = d_zro_deposit_2}
					}
				}
				d_zro_deposit_5 = {
					remove_deposit = d_zro_deposit_5
					random_list = {
						1 = {}
						1 = { add_deposit = d_zro_deposit_1}
