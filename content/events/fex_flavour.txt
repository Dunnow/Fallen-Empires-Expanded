planet_event = {
    id = fex_flavour.14
	hide_window = yes
	is_triggered_only = yes

    immediate = {
		fex_create_exodite_country = yes
		set_owner = last_created_country
		set_capital = yes

		remove_modifier = planet_culture_shock
		remove_building = building_colony_shelter
		add_building = building_fex_exodite_settlements
		add_building = building_fex_runestones

		remove_all_districts = yes

		# Quietly recreate comms with countries so that this isn't lost
		every_country = {
			limit = {
				root = { is_surveyed  = { who = prev status = yes} }
			}
            establish_communications = last_created_country
		}

		last_created_country = {
            set_awareness = 100
		}
		planet_event = { id = fex_flavour.15 } # Regrow natives
		planet_event = { id = fex_flavour.16 } # Get rid of colonists
	}
}

# Exile raid dispatch - select target and send fleet
# Scope = fex_xenophobe_exiles_country
country_event = {
	id = fex_flavour.71
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		set_country_flag = fex_exile_raid_ongoing

		# Select a target empire
		random_country = {
			limit = {
				is_country_type = default
				any_owned_planet = {
					pop_amount > 0
				}
			}
			save_event_target_as = fex_exile_raid_target

			# Select a target planet
			random_owned_planet = {
				limit = { pop_amount > 0 }
				save_event_target_as = fex_exile_raid_planet
				solar_system = {
					save_event_target_as = fex_exile_raid_target_system
				}
			}

			# Establish communications if needed and send first contact event
			if = {
				limit = {
					NOT = { has_communications = root }
				}
				establish_communications_no_message = root
				country_event = { id = fex_flavour.60 }
			}

			# Notify the target
			country_event = { id = fex_flavour.63 }
		}

		# Dispatch the fleet
		random_controlled_fleet = {
			limit = {
				has_fleet_flag = fex_exile_raiding_fleet
				is_in_combat = no
				solar_system = { has_star_flag = fex_xenophobe_exiles_system }
			}
			set_fleet_flag = fex_exile_raid_active
			set_fleet_stance = aggressive
			set_fleet_bombardment_stance = indiscriminate
			set_aggro_range_measure_from = self
			set_aggro_range = 150
			auto_move_to_planet = {
				target = event_target:fex_exile_raid_planet
				clear_auto_move_on_arrival = no
			}
		}
	}
}

# Exile raid bombardment - fired via on_planet_bombarded
# Scope = Planet, From = Bombarder (country)
planet_event = {
	id = fex_flavour.72
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		planet_devastation >= 50
		from = { is_country_type = fex_exiles_country }
		owner = {
			is_country_type = default
		}
		NOT = { has_planet_flag = fex_exile_planet_raided }
	}

	immediate = {
		set_timed_planet_flag = {
			flag = fex_exile_planet_raided
			days = 600
		}

		# Notify the planet owner and apply raid effects
		save_event_target_as = fex_exile_raided_planet
		owner = {
			country_event = { id = fex_flavour.73 }
		}

		# Send the fleet home
		from = {
			random_controlled_fleet = {
				limit = { has_fleet_flag = fex_exile_raid_active }
				fex_end_exile_raid = yes
			}
		}
	}
}
