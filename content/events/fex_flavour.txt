# Exodite world discovered
ship_event = {
    id = fex_flavour.10
    title = fex_flavour.10.title
    desc = fex_flavour.10.desc
    picture = GFX_evt_animal_wildlife
    is_triggered_only = yes
    location = from

    trigger = {
        from = {
            has_planet_flag = fex_exodite_planet
        }
        owner = {
            not = {
                has_country_flag = fex_has_encountered_exodite_world
            }
        }
    }

    immediate = {
        owner = {
            set_country_flag = fex_has_encountered_exodite_world
        }
    }

    option = {
        name = fex_flavour.10.a
        owner = {
            establish_communications = from.owner
            country_event = { id = fex_flavour.11 }
        }
        from.owner = {
            set_awareness = 100
        }
    }
}

# Exodites being invaded.

	id = fex_flavour.12

		FromFrom = {
			is_primitive = yes
		}

			is_fallen_empire_xenophile = yes

		event_target:fex_xenophile_fe = {
            establish_communications = from
        }

	}
}

# Bounce from the planet gatekeeper
# There's technically an edge case here where you could repeatedly invade and remove the pops via land appropriation.
# Grow the pops back? Expel the settlers? Maybe have an event where the population grows back gradually
planet_event = {
    id = fex_flavour.14
	hide_window = yes
	is_triggered_only = yes

    immediate = {
		fex_create_exodite_country = yes
		set_owner = last_created_country
		set_capital = yes

		remove_modifier = planet_culture_shock
		remove_building = building_colony_shelter
		add_building = building_fex_exodite_settlements
		add_building = building_fex_runestones

		remove_all_districts = yes

		# Quietly recreate comms with countries so that this isn't lost
		every_country = {
			limit = {
				root = { is_surveyed  = { who = prev status = yes} }
			}
            establish_communications = last_created_country
		}

		last_created_country = {
            set_awareness = 100
		}
		planet_event = { id = fex_flavour.15 }
	}
}

planet_event = {
    id = fex_flavour.15
	hide_window = yes
	is_triggered_only = yes

    immediate = {
		if = {
			limit = {
				exists = owner
				owner = {
					fex_is_exodite_country = yes
				}
				count_owned_pops = {
					limit = {
						is_same_species = event_target:fex_xenophile_fe.species
					}
					count < 9
				}
			}
			create_pop = {
				species = event_target:fex_xenophile_fe.species
			}
			planet_event = { id = fex_flavour.15 days = 360 }
		}
	}
}
