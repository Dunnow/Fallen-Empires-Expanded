# Initial Setup
event = {
    id = fex_core.1
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        set_global_flag = fallen_empires_expanded_active
        if = {
            limit = { fex_materialist_fe_exists = yes }
            random_country = {
                limit = {  is_fallen_empire_materialist = yes }

            limit = { fex_spiritualist_fe_exists = yes }

                limit = {  is_fallen_empire_machine = yes }
                save_global_event_target_as = fex_machine_fe
            }
        }
    }
}

# Per Country initial setup
country_event = {
    id = fex_core.2
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_fallen_empire = yes
    }

    immediate = {
        # Rename colonies and add some flavour because not naming them is stupid.
        every_owned_planet = {
            if = {
                limit = {
                    fex_is_bespoke_fe_world = no
                }
                set_name = random
                planet_event = { id = fex_core.10 }
            }
            else = {
                set_planet_flag = fex_bespoke_world
            }

        }
        # Give all worlds the precursor environment
        every_owned_planet = {
            if = {
                limit = { is_ringworld = yes }
                add_modifier = { modifier = fex_fe_terraforming_ringworld }
            }
            else_if = {
                limit = { is_planet_class = pc_relic}
                add_modifier = { modifier = fex_decayed_fe_terraforming }
            }
            else_if = {
                limit = { is_planet_class = pc_city}
                add_modifier = { modifier = fex_fe_terraforming_ecu }
            }
            else = {
                add_modifier = { modifier = fex_fe_terraforming }
            }
        }

        # Give uninhabited worlds the ancient terraforming modifier
        every_planet_within_border = {
            limit = {
                is_colonizable = yes
                is_colony = no
            }
            add_modifier = { modifier = fex_decayed_fe_terraforming }
        }

        # Give holy worlds the fe terraforming modifier. Give any other worlds in the system some decayed ones
        every_galaxy_planet = {
            limit = { has_modifier = holy_planet }
            add_modifier = { modifier = fex_fe_terraforming }
            solar_system = {
                every_system_planet = {
                    limit = {
                        is_colonizable = yes
                        NOT = { has_modifier = holy_planet}
                    }
                    add_modifier = { modifier = fex_decayed_fe_terraforming }
                }
            }
        }

        # Set up "historical" spheres.
        every_system = {
            closest_system = {
                max_steps = 5
                limit = {
                    exists = owner
                    owner = {
                        AND = {
                            is_fallen_empire = yes
                            is_fallen_empire_machine = no # The machine FEs are more isolationist.
                        }
                    }
                }
                prev = {
                    set_star_flag = fex_historical_system
                }
                if = {
                    limit = { owner = { is_fallen_empire_spiritualist = yes } }
                    prev = { set_star_flag = fex_spiritualist_history}
                }
                if = {
                    limit = { owner = { is_fallen_empire_materialist = yes } }
                    prev = { set_star_flag = fex_materialist_history}
                }
                if = {
                    limit = { owner = { is_fallen_empire_xenophile = yes } }
                    prev = { set_star_flag = fex_xenophile_history}
                }
                if = {
                    limit = { owner = { is_fallen_empire_xenophobe = yes } }
                    prev = { set_star_flag = fex_xenophobe_history}
                }
            }
        }

        # Give them a hyper relay.
        every_system_within_border  = {
            spawn_megastructure = {
				type = hyper_relay_ruined
				random_pos = yes
                graphical_culture = mammalian_01 # FEs dont have a model for these
			}
            random_system_megastructure = {
                limit = { is_megastructure_type = hyper_relay_ruined }
                upgrade_megastructure_to = hyper_relay_restored
                finish_upgrade = yes
                set_graphical_culture = mammalian_01
            }
        }

        # Spawn random historical systems
        country_event = { id = fex_core.20 } # Ethics-based System gatekeeper
        country_event = { id = fex_core.25 } # Generic Systems

        # Create hyper relay "roads" to points of interest
        # TODO: Refactor this. The recursion just doesnt work. I need a second look at this or it crashes the game
        #country_event = { id = fex_core.3}

        # Add xenophile exodites
        if = {
            limit = { is_fallen_empire_xenophile = yes }
            while = {
                count = 3
                random_galaxy_planet = {
                    limit = {
                        solar_system = {
                            has_star_flag = fex_xenophile_history
                            NOT = { any_system_planet = { is_colonizable = yes } }
                            has_owner = no
                            OR = {
                                is_star_class = sc_f
                                is_star_class = sc_g
                                is_star_class = sc_k
                                is_star_class = sc_m
                            }
                        }
                        is_planet_class = pc_barren
                        not = { has_planet_flag = ancient_history_planet}
                    }
                    fex_generate_exodites_on_planet = yes
                }
            }
        }
    }
}

# Master gatekeeper for road building. Can also build a road by firing fex_core.4 on a system.
event = {
    id = fex_core.3
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        every_system = {
            limit = {
                has_star_flag = fex_historical_system # Would they have been here?
                NAND = {
                    exists = owner
                    owner = { is_fallen_empire = yes}
                }
            }

            if = {
                limit = { # These are location that would always have a road.
                    OR = {
                        AND = {
                            has_any_megastructure = yes
                            fex_has_hyper_relay = no
                            has_no_non_gate_megastructure = no
                        }
                        has_star_flag = fex_spawn_road_source
                    }
                }
                set_star_flag = fex_road_source
                system_event = { id = fex_core.4 }
            }
        }
    }
}

# Follow on road building. Recursively runs until we reach the target.
system_event = {
    id = fex_core.4
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        set_star_flag = fex_road_system
        closest_system = {
            limit = {
                exists = owner
                owner = {
                    AND = {
                        is_fallen_empire = yes
                        is_fallen_empire_machine = no
                    }
                }
            }
            save_event_target_as = fex_road_target
        }
        every_neighbor_system = {
            export_trigger_value_to_variable = {
                variable = fex_road_distance
                trigger = distance
                parameters = { source = event_target:fex_road_target }
            }
        }
        ordered_neighbor_system = {
            order_by = fex_road_distance
            position = 0
            inverse = yes
            # not perfect yet
            limit = {
                not = {
                    has_star_flag = fex_road_system
                    and = {
                        exists = owner
                        owner = { is_regular_empire = yes }
                    }
                }
            }
            if = {
                limit = {
                    NOT = {
                        AND = {
                            exists = owner
                            owner = { is_fallen_empire = yes }
                        }
                    }
                }
                system_event = { id = fex_core.4 }
            }
        }
        if = {
            limit = {
                fex_has_hyper_relay = no
            }
            random_list = {
                10 = {} # Nothing
                40 = {
                    spawn_megastructure = {
                        type = hyper_relay_ruined
                        random_pos = yes
                        graphical_culture = mammalian_01 # FEs dont have a model for these
                        #orbit_angle = 360
                    }
                }
                30 = {
                    spawn_megastructure = {
                        type = hyper_relay
                        random_pos = yes
                        graphical_culture = mammalian_01 # FEs dont have a model for these
                        #orbit_angle = 360
                    }
                }
            }
        }
    }
}

# Xenophobe generic world flavour
planet_event = {
    id = fex_core.11
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        random_list = {
            # Chance of being a unique planet
            10 = {
                random_list = {
                    100 = {
                        fex_set_hunting_world = yes
                        modifier = {
                            factor = 0
                            owner = { any_owned_planet = { has_planet_flag = fex_hunting_world } }
                        }
                    }
                    100 = {
                        fex_set_martial_order_world = yes
                        modifier = {
                            factor = 0
                            owner = { any_owned_planet = { has_planet_flag = fex_martial_order_world } }
                        }
                    }
                    50 = { # Reduced chance because it's generic
                        fex_set_mausoleum_world = yes
                        modifier = {
                            factor = 0
                            owner = { any_owned_planet = { has_planet_flag = fex_mausoleum_world } }
                        }
                    }
                    # Small chance of nothing so we dont get stuck in an infinite loop
                    1 = {
                        remove_planet_flag = fex_special_world
                        set_planet_flag = fex_generic_world
                    }
                }
            }
            20 = { set_planet_flag = fex_generic_world }
        }
    }
}
