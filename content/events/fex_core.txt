namespace = fex_core

# Initial Setup
event = {
    id = fex_core.1
    hide_window = yes
    is_triggered_only = yes

    immediate = {
        set_global_flag = fallen_empires_expanded_active
        if = {
            limit = { fex_materialist_fe_exists = yes }
            random_country = {
                limit = {  is_fallen_empire_materialist = yes }
                save_global_event_target_as = fex_materialist_fe
            }
        }
        if = {
            limit = { fex_spiritualist_fe_exists = yes }
            random_country = {
                limit = {  is_fallen_empire_spiritualist = yes }
                save_global_event_target_as = fex_spiritualist_fe
            }
        }
        if = {
            limit = { fex_xenophobe_fe_exists = yes }
            random_country = {
                limit = {  is_fallen_empire_xenophobe = yes }
                save_global_event_target_as = fex_xenophobe_fe
            }
        }
        if = {
            limit = { fex_xenophile_fe_exists = yes }
            random_country = {
                limit = {  is_fallen_empire_xenophile = yes }
                save_global_event_target_as = fex_xenophile_fe
            }
        }
        if = {
            limit = { fex_machine_fe_exists = yes }
            random_country = {
                limit = {  is_fallen_empire_machine = yes }
                save_global_event_target_as = fex_machine_fe
            }
        }
    }
}

# Per Country initial setup
country_event = {
    id = fex_core.2
    hide_window = yes
    is_triggered_only = yes

    trigger = {
        is_fallen_empire = yes
    }

    immediate = {
        # Rename colonies and add some flavour because not naming them is stupid.
        every_owned_planet = {
            limit = {
                fex_is_bespoke_fe_world = no
            }
            set_name = random
            planet_event = { id = fex_core.10 }
        }
        # Set up "historical" spheres.
        every_system = {
            closest_system = {
                max_steps = 5
                limit = {
                    exists = owner
                    owner = {
                        AND = {
                            is_fallen_empire = yes
                            is_fallen_empire_machine = no # The machine FEs are more isolationist.
                        }
                    }
                }
                prev = {
                    set_star_flag = fex_historical_system
                }
                if = {
                    limit = { owner = { is_fallen_empire_spiritualist = yes } }
                    prev = { set_star_flag = fex_spiritualist_history}
                }
                if = {
                    limit = { owner = { is_fallen_empire_materialist = yes } }
                    prev = { set_star_flag = fex_materialist_history}
                }
                if = {
                    limit = { owner = { is_fallen_empire_xenophile = yes } }
                    prev = { set_star_flag = fex_xenophile_history}
                }
                if = {
                    limit = { owner = { is_fallen_empire_xenophobe = yes } }
                    prev = { set_star_flag = fex_xenophobe_history}
                }
            }
        }

        # Give them a hyper relay.
        every_system_within_border  = {
            spawn_megastructure = {
				type = hyper_relay
				random_pos = yes
                graphical_culture = mammalian_01 # FEs dont have a model for these
                #orbit_angle = 360
			}
        }

        # Create hyper relay "roads" to points of interest
        # TODO: Refactor this. The recursion just doesnt work. I need a second look at this or it crashes the game
        #country_event = { id = fex_core.3}

        # Add xenophile exodites
        if = {
            limit = { is_fallen_empire_xenophile = yes }
            while = {
                count = 3
                random_galaxy_planet = {
                    limit = {
                        solar_system = {
                            has_star_flag = fex_xenophile_history
                            NOT = { any_system_planet = { is_colonizable = yes } }
                            has_owner = no
                            OR = {
                                is_star_class = sc_f
                                is_star_class = sc_g
                                is_star_class = sc_k
                                is_star_class = sc_m
                            }
                        }
                        is_planet_class = pc_barren
                        not = { has_planet_flag = ancient_history_planet}
                    }
                    fex_generate_exodites_on_planet = yes
                }
            }
        }
    }
}

# Master gatekeeper for road building. Can also build a road by firing fex_core.4 on a system.
event = {
    id = fex_core.3
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        every_system = {
            limit = {
                has_star_flag = fex_historical_system # Would they have been here?
                NAND = {
                    exists = owner
                    owner = { is_fallen_empire = yes}
                }
            }

            if = {
                limit = { # These are location that would always have a road.
                    OR = {
                        AND = {
                            has_any_megastructure = yes
                            fex_has_hyper_relay = no
                            has_no_non_gate_megastructure = no
                        }
                        has_star_flag = fex_spawn_road_source
                    }
                }
                set_star_flag = fex_road_source
                system_event = { id = fex_core.4 }
            }
        }
    }
}

# Follow on road building. Recursively runs until we reach the target.
system_event = {
    id = fex_core.4
    is_triggered_only = yes
    hide_window = yes

    immediate = {
        set_star_flag = fex_road_system
        closest_system = {
            limit = {
                exists = owner
                owner = {
                    AND = {
                        is_fallen_empire = yes
                        is_fallen_empire_machine = no
                    }
                }
            }
            save_event_target_as = fex_road_target
        }
        every_neighbor_system = {
            export_trigger_value_to_variable = {
                variable = fex_road_distance
                trigger = distance
                parameters = { source = event_target:fex_road_target }
            }
        }
        ordered_neighbor_system = {
            order_by = fex_road_distance
            position = 0
            inverse = yes
            # not perfect yet
            limit = {
                not = {
                    has_star_flag = fex_road_system
                    and = {
                        exists = owner
                        owner = { is_regular_empire = yes }
                    }
                }
            }
            if = {
                limit = {
                    NOT = {
                        AND = {
                            exists = owner
                            owner = { is_fallen_empire = yes }
                        }
                    }
                }
                system_event = { id = fex_core.4 }
            }
        }
        if = {
            limit = {
                fex_has_hyper_relay = no
            }
            random_list = {
                10 = {} # Nothing
                40 = {
                    spawn_megastructure = {
                        type = hyper_relay_ruined
                        random_pos = yes
                        graphical_culture = mammalian_01 # FEs dont have a model for these
                        #orbit_angle = 360
                    }
                }
                30 = {
                    spawn_megastructure = {
                        type = hyper_relay
                        random_pos = yes
                        graphical_culture = mammalian_01 # FEs dont have a model for these
                        #orbit_angle = 360
                    }
                }
            }
        }
    }
}
